\section{Start}

\begin{korollar}{1.16}
    \label{korollar:inf-sup}
    $A \subseteq B \subseteq \R$, dann
    \begin{itemize}
        \item $B$ nach oben beschraenkt $\implies \sup A \le \sup B$
        \item $B$ nach unten beschraenkt $\implies \inf B \le \inf A$
    \end{itemize}
\end{korollar}
